    "def evaluate_model(name, y_true, y_pred, y_prob):\n",
    "    print(f\"--- {name} ---\")\n",
    "    print(\"Accuracy :\", accuracy_score(y_true, y_pred))\n",
    "    print(\"Precision:\", precision_score(y_true, y_pred))\n",
    "    print(\"Recall   :\", recall_score(y_true, y_pred))\n",
    "    print(\"F1-Score :\", f1_score(y_true, y_pred))\n",
    "    print(\"ROC-AUC  :\", roc_auc_score(y_true, y_prob))\n",
    "    print()\n",
    "\n",
    "    cm = confusion_matrix(y_true, y_pred)\n",
    "    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues')\n",
    "    plt.title(f'Confusion Matrix - {name}')\n",
    "    plt.xlabel('Predicted')\n",
    "    plt.ylabel('Actual')\n",
    "    plt.show()\n",
    "\n",
    "log_prob = log_model.predict_proba(X_test)[:, 1]\n",
    "rf_prob = rf_model.predict_proba(X_test)[:, 1]\n",
    "\n",
    "evaluate_model(\"Logistic Regression\", y_test, log_preds, log_prob)\n",
    "evaluate_model(\"Random Forest\", y_test, rf_preds, rf_prob)\n"
